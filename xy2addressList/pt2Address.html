h = httplib2.Http(timeout=None)

def gpd2Address(row):
    Road_Index = row[1]['Road_Index']
    Road_Name = row[1]['Road_Name']
    x = str(row[1]['geometry'].x)
    y = str(row[1]['geometry'].y)

    xy = y+','+x
    resUrl = f'https://gist.motc.gov.tw/gist_api/V2/Map/GeoCode/Address/{xy}?$format=GEOJSON'
    res, content = h.request(resUrl)
    address = json.loads(str(content, encoding = "utf-8"))
    return [Road_Index,Road_Name,xy,address]
